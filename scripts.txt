create table publications(
	id_publication varchar(3) primary key not null,
	title varchar(100) not null,
	pdf_publication varchar(100) not null,
	img_publication varchar(100) not null,
	abstract varchar(500) not null, 
	codigo varchar(100),
	url varchar(100),
	date_publication date
#DATE - format YYYY-MM-DD
	);

create table author(
	id_publication varchar(3),
	id_people varchar(3),
    CONSTRAINT id_persons FOREIGN KEY (id_people) REFERENCES people_profiles(id_profile),
    CONSTRAINT id_pubs FOREIGN KEY (id_publication) REFERENCES publications(id_publication)
);
